Step 2: Create Post-Chat Indicator

File: src/services/elementChatIndicatorService.ts

const INDICATOR_SIZE = {
  width: 30,   // 30% of collapsed size
  height: 30
};

function createPostChatIndicator(
  element: HTMLElement,
  sessionId: string,
  state: ElementChatState
): HTMLElement {
  const indicator = document.createElement('div');
  indicator.className = 'nabokov-post-chat-indicator';
  indicator.dataset.sessionId = sessionId;

  // Styling
  Object.assign(indicator.style, {
    position: 'absolute',
    width: `${INDICATOR_SIZE.width}px`,
    height: `${INDICATOR_SIZE.height}px`,
    backgroundColor: STATE_COLORS[state],
    borderRadius: '4px',
    border: '2px solid #d4a259',
    boxShadow: '0 2px 8px rgba(177, 64, 60, 0.3)',
    cursor: 'pointer',
    zIndex: '9999',
    transition: 'all 0.2s ease'
  });

  // Position at top-right
  updatePosition(indicator, element);

  // Click to reopen
  indicator.addEventListener('click', () => {
    elementChatWindowManager.openWindow(sessionId);
  });

  // Hover effect
  indicator.addEventListener('mouseenter', () => {
    indicator.style.transform = 'scale(1.1)';
  });

  indicator.addEventListener('mouseleave', () => {
    indicator.style.transform = 'scale(1)';
  });

  document.body.appendChild(indicator);
  return indicator;
}

Step 3: Update on Window Close

File: src/services/elementChatWindowManager.ts

export function closeWindow(sessionId: string) {
  const session = sessions.get(sessionId);
  if (!session) return;

  // Show post-chat indicator
  // Skip for text-selection chats (they have inline highlights)
  if (session.selectionType !== 'text-selection') {
    elementChatIndicatorService.showPostChatIndicator(
      session.element,
      sessionId,
      'active'
    );
  }

  // Remove window
  windows.get(sessionId)?.remove();
  windows.delete(sessionId);
}

const [isStreaming, setIsStreaming] = useState(false);

const handleSend = async (message: string) => {
  setIsStreaming(true);

  // Update indicator to loading state
  elementChatIndicatorService.updateIndicatorState(sessionId, 'loading');

  try {
    await streamChatResponse(sessionId, message, {
      onChunk: (chunk) => { /* Update UI */ },
      onComplete: () => {
        setIsStreaming(false);
        elementChatIndicatorService.updateIndicatorState(sessionId, 'active');
      },
      onError: (error) => {
        setIsStreaming(false);
        elementChatIndicatorService.updateIndicatorState(sessionId, 'error');
      }
    });
  } catch (error) {
    setIsStreaming(false);
    elementChatIndicatorService.updateIndicatorState(sessionId, 'error');
  }
};
